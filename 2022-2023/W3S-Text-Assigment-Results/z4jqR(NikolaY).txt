0 MySQL SELECТ - връща информация от таблицата (от базата данни)
1 MySQL WHERE - условие за филтър
0 MySQL AND, OR, NOT - логични операции (оператори)
1 MySQL ORDER BY - сортира ги по условие
1 MySQL INSERT INTO - вкарва нови редове в таблицата
0 MySQL NULL Values - редове, които нямат стойност (не само редове)
0 MySQL UPDATE - ъпдейтва редове (не само редове)
0 MySQL DELETE - изтрива редове (не само редове)
1 MySQL LIMIT - лимитира колко колони да се върнат
1 MySQL MIN and MAX - връщат най-малката и най-голямата стойност от колоната
1 MySQL COUNT - броя на редовете, които съвпадат с дадено условие
        AVG - средната стойност на подадените редове
        SUM - сборът от редовете
1 MySQL LIKE - сравнява текст
0 MySQL Wildcards - шаблон за стринг (не е подробно)
0 MySQL IN - могат да се подадат няколко стойност и да се сравни с тях (не е подробно)
0 MySQL BETWEEN - като IN, но между две стойности (не е подробно)
0 MySQL Aliases - дава име на колоната,таблицата или ред (временно име)
1 MySQL Joins - комбинира редове от две+ таблици
1 MySQL INNER JOIN - имат общи стойности и в двете таблици
1 MySQL LEFT JOIN - всичко от лявата таблица и общите стойности
1 MySQL RIGHT JOIN - всичко от дясната таблица и общите стойности
0 MySQL DISTINCT - връща един ред, използва се в WHERE
0 MySQL GROUP BY - групира (как)
1 MySQL Create DB - създаване на датабаза
1 MySQL Drop DB - изтрива датабаза
1 MySQL Create Table - създава таблица
1 MySQL Drop Table - изтрива таблица
0 MySQL Alter Table - добавя колона в таблицата (променя данни)
1 MySQL Constraints - при създаване на таблицата, се задават лимити за колоните
1 MySQL Not Null - стойността да не е NULL
1 MySQL Unique - стойността да е единствена
0 MySQL Primary Key - стоността, която се използва за индексиране
1 MySQL Foreign Key - използва се за свързване с други таблици
0 MySQL Check - наша проверка на която стойността трябва да отговаря (нищо не разбрах)
0 MySQL Default - стойността по подразбиране (не е подробно)
0 MySQL Auto Increment - стойността се генерира автоматично (не е подробно)
