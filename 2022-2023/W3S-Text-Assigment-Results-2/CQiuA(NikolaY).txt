1 MySQL SELECT - избираме кои колони да се покажат от дадена таблица
1 MySQL WHERE - условие, по което да се показват дадени записи от таблицата
1 MySQL AND, OR, NOT - логически операции, които се използват при WHERE(ОПЕРАТОРИ)
1 MySQL ORDER BY - по коя колона да се сортират данните
1 MySQL INSERT INTO - да се добави запис в дадена таблица
1 MySQL NULL Values - нямат стойност
1 MySQL UPDATE - update на запис
1 MySQL DELETE - delete а запис
1 MySQL LIMIT - лимитиране на колко записа да се покажат
1 MySQL MIN and MAX - показват минимална/максимална от дадена колона
1 MySQL COUNT, AVG, SUM - изпозва се за броене, средноаритметично, сума на стойности от колона
1 MySQL LIKE - изпозва се за сравнение на стрингове
1 MySQL Wildcards - използва се в LIKE, за да се намират подобни стрингове. Например %
0 MySQL IN - проверка дали стойности се намират някъде (таблица) (не споменаваш нищо за where)
1 MySQL BETWEEN - между две стойности
1 MySQL Aliases - временно наименование на колона
1 MySQL Joins - присъединяване на таблица по някаква обща/matching колона
1 MySQL INNER JOIN - изкарва matching стойности
1 MySQL LEFT JOIN - изкарва всичко от лявата и matching стойности от дясната (АБЕ КАК МОЕ ТОЛКО ДА ПРЕПИСВАШ БЕ EZ AKADEMI4NA)
1 MySQL RIGHT JOIN - изкарва всичко от дясната и matching стойности от лявата
1 MySQL CROSS JOIN - изкарва всички стойности и от двете
1 MySQL Self Join - предполагам таблица е join-ата със себе си  
0 MySQL UNION - не сме го учили
1 MySQL DISTINCT - същото като UNIQUE ама за query
1 MySQL GROUP BY - групира по дадена колона

1 MySQL Database - датабаза, в която се добавят таблици
1 MySQL Create DB - създаване на датабаза
1 MySQL Drop DB - да се махне датабаза  
1 MySQL Create Table - създаване на таблица
1 MySQL Drop Table - да се махне таблица
1 MySQL Alter Table - промяна на таблица
1 MySQL Constraints - ограничения
1 MySQL Not Null - не-празни стойности. Изпозва се при WHERE (използва се не само при where)
1 MySQL Unique - стойност да е уникална/да не се повтаря
1 MySQL Primary Key - уникален ключ в дадена таблица (ID)
1 MySQL Foreign Key - връзка към PRIMARY KEY на друга таблица
1 MySQL Check - проверява за стойности
1 MySQL Default - ако не се подаде стойност при добавяне на запис, каква да бъде тази стойност
1 MySQL Auto Increment - със всяко добавяне автоматично да се увеличава с едно. Използва се при PRIMARY KEY
