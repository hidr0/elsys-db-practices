0 MySQL SELECT - принтирана избрана колона "ne printira, a vzima informaciq"

0 MySQL WHERE - условие, което се попълва и в него се слага условие "klauza za filtraciq na stoinosti" 

1 MySQL AND, OR, NOT - логически операции и, или, не

1 MySQL ORDER BY - подреди, по дедофт е във възходящ ред

1 MySQL INSERT INTO - доави в таблицата данни

0 MySQL NULL Values - нулирай данните "pole bez stoinost"

1 MySQL UPDATE - промени избраните полета, не добавя нови неща, променя стари

0 MySQL DELETE - изтрий "kakvo?krushi? stoinosti ot tablica"

0 MySQL LIMIT - илмитай, слагат се някакви граници (?) " kolko stoinosti da vurne"

1 MySQL MIN and MAX - намира минималната и максимална стойност в колоната

1 MySQL COUNT, AVG, SUM - преброй, средно аритметично, сбор в колоната

1 MySQL LIKE - оператор за сравняване на стрингове

0 MySQL Wildcards - не схванах "w3schools"

1 MySQL IN - в нещо си - изброява се множество от елементи от дадена колона и се търси в това множество

1 MySQL BETWEEN - между, тази и тази стойност

0 MySQL Aliases - не схванах "drugo ime za kolona ili tablica"

1 MySQL Joins - може би това е общото наименование на различните join-ове (?) - те хващат две или повече таблици и ги лепят заедно

0 MySQL INNER JOIN - вътрешен join "изкарва matching стойности"

0 MySQL LEFT JOIN - ляв join "изкарва всичко от лявата и matching стойности от дясната"

0 MySQL RIGHT JOIN - десен join "изкарва всичко от дясната и matching стойности от лявата"

0 MySQL CROSS JOIN - някакъв странен join "изкарва всички стойности и от двете"

0 MySQL Self Join - нямам идея "предполагам таблица е join-ата със себе си "

0 MySQL UNION - не знам "ne sme go uchili"

1 MySQL DISTINCT - като unique ама за куерита

1 MySQL GROUP BY - групирай по (условие)


0 MySQL Database - йес да

1 MySQL Create DB - създай датабаза

1 MySQL Drop DB - дропни базата

1 MySQL Create Table - създай таблица

1 MySQL Drop Table - дропни таблицата

0 MySQL Alter Table - не помня "promenq tablica"

0 MySQL Constraints - не знам "ogranicheniq"

1 MySQL Not Null - при създаване на поле задава условието да не е нула

1 MySQL Unique - при създаване на поле задава, то да бъде уникално и да не се повтаря от друго

1 MySQL Primary Key - основен ключ (индекс)

1 MySQL Foreign Key - външен ключ от друга таблица 

1 MySQL Check - проверява стойността

1 MySQL Default - задава се као не се въведе нищо да има стойност, която се попълва там автоматично

0 MySQL Auto Increment - номерирай автоматично "vdiga se s 1"
