0 MySQL SELECT - взима информацията от дадена таблица (от базата данни)
1 MySQL WHERE - за проверка на данни
1 MySQL AND, OR, NOT - използват се в условие (AND - И, OR - ИЛИ, NOT - НЕ ТОВА)
1 MySQL ORDER BY - подрежда в възходящ или низходящ ред
1 MySQL INSERT INTO - вкарваме стойности в таблицата
0 MySQL NULL Values - представляват невъведено поле в таблицата(ТОВА НЕСЪШЕСТВЪВАЩИ НЕЩАААЯ)
1 MySQL UPDATE - ъпдейтваме таблицата с нови стойности
0 MySQL DELETE - изтриване на ред от таблицата(не само редове)
1 MySQL LIMIT - лимитира до определен брой стойности
1 MySQL MIN and MAX - най-малка и най-голяма стойност
0 MySQL COUNT, AVG, SUM - преброява стойностите за един ред, намира средната стойност, сумира стойностите(научи се кво е ред и кво е колона)
0 MySQL LIKE - проверява дали са еднакви две стойности(бачка със стрингове)
0 MySQL Wildcards - идея си нямам
0 MySQL IN - проверява дали в дадена стойност е част от стойностите, които са му дадени
0 MySQL BETWEEN - идея си нямам
0 MySQL Aliases - идея си нямам
0 MySQL Joins - съединява всички елементи от две таблици(КВИ ВСИЧКИ ЕЛЕМЕНТИ БЕЕЕ)
1 MySQL INNER JOIN - съединява две таблици като взима само свързаните елементи от двете таблици
1 MySQL LEFT JOIN - съединява две таблици като взима всички елементи от първата таблица и залепя за тях елементите свързани с първата таблица от втората таблица
1 MySQL RIGHT JOIN - съединява две таблици като взима всички елементи от втората таблица и залепя за тях елементите свързани с втората таблица от първата таблица
0 MySQL CROSS JOIN - идея си нямам
0 MySQL Self Join - идея си нямам
0 MySQL UNION - залепва два резултата от query(какво куери)
1 MySQL DISTINCT - връща само по една от няколко еднакви стойности
1 MySQL GROUP BY - групира стойности при определено условие

1 MySQL Database - съхранява всички таблици и информация
0 MySQL Create DB - създава датабазата ако не съществува(независимо дали съществува)
1 MySQL Drop DB - изтрива датабазата
1 MySQL Create Table - създава таблица
1 MySQL Drop Table - изтрива таблица
0 MySQL Alter Table - добавяне на редове при вече създадена таблица(променя таблицата)
0 MySQL Constraints - идея си нямам
1 MySQL Not Null - стойността да не е нулева
1 MySQL Unique - стойността да е уникална
1 MySQL Primary Key - Главният ключ в таблицата (id)
1 MySQL Foreign Key - Ключ с който се свързва с главният ключ на друга таблица(главния*)
1 MySQL Check - проверяваме стойостите при дадено условие
1 MySQL Default - предполагам че задава дефоутна стойност
1 MySQL Auto Increment - автоматично да се увеличава
