1 MySQL SELECT - изкарва информация за дадената таблица
0 MySQL WHERE - използва се за условие(Използва се да филтрира) 
1 MySQL AND, OR, NOT - и , или , не операции
0 MySQL ORDER BY - подрежда по дадена колона(подрежда данни в нисходящ или възходящ ред) 
1 MySQL INSERT INTO - добавя нов запис в таблицата
1 MySQL NULL Values - празна стойност
1 MySQL UPDATE - актуализира даден запис
1 MySQL DELETE - изтрива даден запис
1 MySQL LIMIT - дава се някакъв лимит за редове които да се покажат
0 MySQL MIN and MAX - намира минималната и максималната стойност на даденото поле като минава през всички записи (намира минималната или максималната стойност в дадена колона)
1 MySQL COUNT, AVG, SUM - брой, средно аритметично и сума на дадено поле
1 MySQL LIKE - сравнение на стринг
0 MySQL Wildcards - не знам (използват се за замяна на един или повече символа в стринг)
1 MySQL IN - проверява дали стойността на полето е в изброените стойности
1 MySQL BETWEEN - проверява дали стойност на полето е между две други 
0 MySQL Aliases - не знам(дава временно име на таблица или колона от таблица)
0 MySQL Joins - джойнва 2 таблици (комбинира редовете в две или повече таблици според обща колона)
1 MySQL INNER JOIN - джойнва общите записки на 2 таблици
1 MySQL LEFT JOIN - джойнва всички записки от ляво със съвпадащите от дясно
1 MySQL RIGHT JOIN -  джойнва всички записки от дясно със съвпадащите от ляво
1 MySQL CROSS JOIN - джойнва всички записки от ляво и от дясно
0 MySQL Self Join - не знам (джойн в който таблицата е джойнната със себе си)
0 MySQL UNION - обединява 2 таблици (съединява Selectи)
0 MySQL DISTINCT - не знам (променливата която е дистинкт трябва да има различна стойност при всяка своя прочетена инстанция)
0 MySQL GROUP BY - групира по колони (групира полета от колони според условие)

1 MySQL Database - база данни
1 MySQL Create DB - създава база данни
1 MySQL Drop DB - унищожава база данни
1 MySQL Create Table - създава таблица
1 MySQL Drop Table - унищожава таблица
1 MySQL Alter Table -  променя таблицата
0 MySQL Constraints - не знам (използва се за специфициране на правила за таблицата)
1 MySQL Not Null - не е празна стойност
0 MySQL Unique - уникално(усигорява уникалност на колона)
1 MySQL Primary Key - уникален ключ
0 MySQL Foreign Key - свързва две таблици (поле или сбор от полета от таблицата които реферират Primary Key на друга таблица )
0 MySQL Check - проверява дали стойност е в колона(използва се за лимитиране на стойностите на дадена колона)
1 MySQL Default - стойност по подразбиране
1 MySQL Auto Increment - инкрементира се автоматично
